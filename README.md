for i in range(len(word) // 2):   # 입력받은 문자 길이의 절반까지 반복
  if word[i]  != word[-1 - i]:    # 단어의 맨 처음과 맨 마지막부터 비교
    is_palindrome = False         # 서로 다르면 회문이 아니므로 False
    break
